  mutate(html = glue(template)) %>% 
  mutate(trans_zab = log2(Заболело),

  leaflet() %>% 
  addTiles() %>% 
  addCircleMarkers(lng = ~Долгота, lat = ~Широта, popup = ~html, color = ~ pal, radius = ~ log(Умерло))
